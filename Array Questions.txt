# 1. Reverse an Array
def reverse_array(arr):
    left = 0
    right = len(arr) - 1
    while left < right:
        arr[left], arr[right] = arr[right], arr[left]
        left += 1
        right -= 1
    return arr


# 2. Find the Maximum & Minimum Element
def find_max_min(arr):
    maximum = arr[0]
    minimum = arr[0]
    for num in arr:
        if num > maximum:
            maximum = num
        if num < minimum:
            minimum = num
    return maximum, minimum


# 3. Find the Sum of Elements
def find_sum(arr):
    total = 0
    for num in arr:
        total += num
    return total


# 4. Find the Second Largest Element
def second_largest(arr):
    largest = float('-inf')
    second = float('-inf')
    for num in arr:
        if num > largest:
            second = largest
            largest = num
        elif num > second and num != largest:
            second = num
    return second


# 5. Count Frequency of Elements
def count_frequency(arr):
    freq = {}
    for num in arr:
        if num in freq:
            freq[num] += 1
        else:
            freq[num] = 1
    return freq


# 6. Check if Array is Sorted
def is_sorted(arr):
    for i in range(len(arr) - 1):
        if arr[i] > arr[i + 1]:
            return False
    return True


# 7. Rotate Array by k Positions (Right Rotation)
def rotate_array(arr, k):
    n = len(arr)
    k = k % n
    return arr[-k:] + arr[:-k]


# 8. Find Pair with Given Sum
def find_pair(arr, target):
    seen = set()
    for num in arr:
        complement = target - num
        if complement in seen:
            return (complement, num)
        seen.add(num)
    return None


# 9. Remove Duplicates from Array Maintain Order as well
def remove_duplicates(arr):
    seen = set()
    result = []
    for num in arr:
        if num not in seen:
            result.append(num)
            seen.add(num)
    return result


# 10. Merge Two Sorted Arrays
def merge_sorted_arrays(arr1, arr2):
    i = j = 0
    merged = []
    while i < len(arr1) and j < len(arr2):
        if arr1[i] < arr2[j]:
            merged.append(arr1[i])
            i += 1
        else:
            merged.append(arr2[j])
            j += 1
    while i < len(arr1):
        merged.append(arr1[i])
        i += 1
    while j < len(arr2):
        merged.append(arr2[j])
        j += 1
    return merged


# 11. Remove Given Element from Array
def remove_element(arr, value):
    index = 0
    for i in range(len(arr)):
        if arr[i] != value:
            arr[index] = arr[i]
            index += 1
    return arr[:index]


# 12. Find the Missing Number (1 to n)
def find_missing_number(arr, n):
    expected_sum = n * (n + 1) // 2
    actual_sum = sum(arr)
    return expected_sum - actual_sum


# 13. Find Duplicates in an Array
def find_duplicates(arr):
    seen = set()
    duplicates = set()
    for num in arr:
        if num in seen:
            duplicates.add(num)
        else:
            seen.add(num)
    return list(duplicates)

# 14. Intersection of Two Arrays
def intersection(arr1, arr2):
    result = []
    for i in range(len(arr1)):
        for j in range(len(arr2)):
            if arr1[i] == arr2[j]:
                # check duplicate
                found = False
                for k in range(len(result)):
                    if result[k] == arr1[i]:
                        found = True
                        break
                if not found:
                    result.append(arr1[i])
    return result


# 15. Union of Two Arrays
def union(arr1, arr2):
    result = []

    # add arr1 elements
    for i in range(len(arr1)):
        found = False
        for j in range(len(result)):
            if result[j] == arr1[i]:
                found = True
                break
        if not found:
            result.append(arr1[i])

    # add arr2 elements
    for i in range(len(arr2)):
        found = False
        for j in range(len(result)):
            if result[j] == arr2[i]:
                found = True
                break
        if not found:
            result.append(arr2[i])

    return result


# 16. Check if Two Arrays Are Equal
def are_equal(arr1, arr2):
    if len(arr1) != len(arr2):
        return False

    visited = [False] * len(arr2)

    for i in range(len(arr1)):
        found = False
        for j in range(len(arr2)):
            if arr1[i] == arr2[j] and not visited[j]:
                visited[j] = True
                found = True
                break
        if not found:
            return False
    return True


# 17. Leader Elements
def leaders(arr):
    n = len(arr)
    result = []

    max_right = arr[n - 1]
    result.append(max_right)

    for i in range(n - 2, -1, -1):
        if arr[i] > max_right:
            result.append(arr[i])
            max_right = arr[i]

    # reverse manually
    for i in range(len(result)//2):
        temp = result[i]
        result[i] = result[len(result)-1-i]
        result[len(result)-1-i] = temp

    return result


# 18. Move Zeroes to End
def move_zeros(arr):
    n = len(arr)
    result = [0] * n
    index = 0

    for i in range(n):
        if arr[i] != 0:
            result[index] = arr[i]
            index += 1

    return result


# 19. Subarray with Given Sum (positive numbers)
def subarray_sum(arr, target):
    start = 0
    curr_sum = 0

    for end in range(len(arr)):
        curr_sum += arr[end]

        while curr_sum > target and start <= end:
            curr_sum -= arr[start]
            start += 1

        if curr_sum == target:
            result = []
            for i in range(start, end + 1):
                result.append(arr[i])
            return result

    return []


# 20. Rotate Array Left by k
def rotate_left(arr, k):
    n = len(arr)
    k = k % n

    result = [0] * n
    index = 0

    for i in range(k, n):
        result[index] = arr[i]
        index += 1

    for i in range(0, k):
        result[index] = arr[i]
        index += 1

    return result


# 21. Kth Smallest Element (Selection Sort approach)
def kth_smallest(arr, k):
    n = len(arr)

    # selection sort
    for i in range(n):
        min_idx = i
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j

        temp = arr[i]
        arr[i] = arr[min_idx]
        arr[min_idx] = temp

    return arr[k - 1]


# 22. Find All Subarrays
def all_subarrays(arr):
    result = []

    for i in range(len(arr)):
        for j in range(i, len(arr)):
            sub = []
            for k in range(i, j + 1):
                sub.append(arr[k])
            result.append(sub)

    return result


# 23. Maximum Sum Subarray (Kadane’s Algorithm)
def max_subarray(arr):
    max_sum = arr[0]
    curr_sum = arr[0]

    for i in range(1, len(arr)):
        if curr_sum + arr[i] > arr[i]:
            curr_sum = curr_sum + arr[i]
        else:
            curr_sum = arr[i]

        if curr_sum > max_sum:
            max_sum = curr_sum

    return max_sum
